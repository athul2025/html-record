public class StringManipulation { public static void main(String[] args) { String str1 = "Hello"; String str2 = "World"; String str3 = "hello"; // length() System.out.println("Length of str1: " + str1.length()); // concat() System.out.println("Concat: " + str1.concat(" " + str2)); // charAt() System.out.println("Character at index 1: " + str1.charAt(1)); // getChars() char[] ch = new char[5]; str1.getChars(0, 5, ch, 0); System.out.println("getChars result: " + new String(ch)); // compareTo() System.out.println("CompareTo: " + str1.compareTo(str2)); // equals() System.out.println("Equals: " + str1.equals(str3)); // equalsIgnoreCase() System.out.println("Equals Ignore Case: " + str1.equalsIgnoreCase(str3)); // startsWith() System.out.println("StartsWith 'He': " + str1.startsWith("He")); // endsWith() System.out.println("EndsWith 'lo': " + str1.endsWith("lo")); // contains() System.out.println("Contains 'ell': " + str1.contains("ell")); // indexOf() System.out.println("IndexOf 'l': " + str1.indexOf('l')); // toUpperCase() System.out.println("Uppercase: " + str1.toUpperCase()); // toLowerCase() System.out.println("Lowercase: " + str2.toLowerCase()); // replace() System.out.println("Replace l with x: " + str1.replace('l', 'x')); // Type conversions Double d = Double.valueOf("123.45"); Integer i = Integer.valueOf("100"); System.out.println("Double value: " + d); System.out.println("Integer value: " + i); } }
